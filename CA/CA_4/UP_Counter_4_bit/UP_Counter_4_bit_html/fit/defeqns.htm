********** Mapped Logic **********
FDCPE_Q0: FDCPE port map (Q0,Q0_D,clk,'0','0','1');
     Q0_D <= (NOT Q0 AND NOT reset);
FDCPE_Q1: FDCPE port map (Q1,Q1_D,clk,'0','0','1');
     Q1_D <= ((Q0 AND NOT reset AND NOT Q1)
      OR (NOT Q0 AND NOT reset AND Q1));
FTCPE_Q2: FTCPE port map (Q2,Q2_T,clk,'0','0','1');
     Q2_T <= ((reset AND Q2)
      OR (Q0 AND NOT reset AND Q1));
FTCPE_Q3: FTCPE port map (Q3,Q3_T,clk,'0','0','1');
     Q3_T <= ((reset AND Q3)
      OR (Q0 AND NOT reset AND Q1 AND Q2));
